CREATE TABLE project (
  pro_id      NUMBER NOT NULL,
  emp_id      NUMBER NOT NULL,
  com_id      NUMBER NOT NULL,
  pro_name    VARCHAR2(100) NOT NULL,
  pro_desc    VARCHAR2(1000),
  pro_status  VARCHAR2(20) DEFAULT 'todo' NOT NULL,
  start_date  DATE NOT NULL,
  end_date    DATE,
  created_at  DATE DEFAULT SYSDATE,
  updated_at  DATE DEFAULT SYSDATE,
  CONSTRAINT pk_project PRIMARY KEY (pro_id),
  CONSTRAINT ck_project_status CHECK (pro_status IN ('TODO','DOING','DONE')),
  CONSTRAINT fk_project_employee1 FOREIGN KEY (emp_id)
    REFERENCES employee (emp_id),
  CONSTRAINT fk_project_company1 FOREIGN KEY (com_id)
    REFERENCES company (com_id)
);

CREATE SEQUENCE seq_project START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE;

CREATE OR REPLACE TRIGGER trg_project_bi
BEFORE INSERT ON project
FOR EACH ROW
BEGIN
  IF :NEW.pro_id IS NULL THEN
    SELECT seq_project.NEXTVAL INTO :NEW.pro_id FROM dual;
  END IF;
END;
/

CREATE OR REPLACE TRIGGER trg_project_bu
BEFORE UPDATE ON project
FOR EACH ROW
BEGIN
  :NEW.updated_at := SYSDATE;
END;
/

CREATE INDEX fk_project_employee1_idx ON project (emp_id);
CREATE INDEX fk_project_company1_idx  ON project (com_id);

 ## 1. AI 기반 프로젝트 리스크 예측 & 자동 알림

> **실무진의 고민:** 프로젝트가 지연되면 비용·인력·고객 신뢰에 직격탄이 됩니다. 하지만 대부분은 문제가 발생한 뒤에야 알게 되어 대응이 늦습니다.

* **아이디어:** 프로젝트 진행률, 지연 태스크, 남은 일정 데이터를 AI(OpenAI API)로 분석하여 **지연 가능성을 사전에 예측**합니다. 위험도가 높을 경우 Slack/Discord Webhook으로 관리자와 팀원에게 즉시 알림을 발송합니다.  
  예: “프론트엔드 모듈은 예상 완료일보다 3일 늦어질 가능성이 70%입니다. 백업 인력 투입을 권장합니다.”
* **API 및 기술적 구현:**
  * **OpenAI API:** 프로젝트 데이터(진행률, 태스크 상태, 일정)를 입력 → 지연 가능성 및 리스크 점수 산출.
  * **Slack/Discord Webhook:** 예측 결과를 팀 채널에 실시간 전송.
  * **추가 포인트:** 알림에 **대안 액션 제안**(예: 인력 재배치, 일정 조정)을 포함하면 단순 경고가 아니라 **실질적 의사결정 지원 도구**가 됩니다.
* **풀스택 어필 포인트:** 단순 CRUD를 넘어서 **데이터 분석 + 외부 협업 툴 연동**을 구현했다는 점을 보여줍니다.
* **면접관이 열광하는 이유:** “이 개발자는 단순 관리가 아니라 **문제 예방**까지 고민했네”라는 인상을 줍니다. 실제로 PM들이 가장 원하는 기능 중 하나입니다.

## 2. 태스크 자동 요약 & 주간 보고서 생성

> **실무진의 고민:** 팀장은 매주 수십 개 태스크를 취합해 보고서를 작성해야 합니다. 반복적이고 시간이 많이 소모되는 작업입니다.

* **아이디어:** 태스크 완료/진행 상황을 AI가 요약하여 **자동 주간 보고서**를 생성합니다.  
  예: “이번 주 완료 태스크 12건, 지연 태스크 3건, 주요 리스크: 백엔드 API 응답 지연”
* **API 및 기술적 구현:**
  * **OpenAI API:** 태스크 데이터(완료/진행/지연)를 입력 → 자연어 요약 생성.
  * **Google Docs API:** 자동으로 보고서를 문서화하여 공유 가능한 형태로 저장.
  * **추가 포인트:** 보고서에 **리스크 하이라이트**와 **다음 주 우선순위 제안**을 포함하면 단순 요약이 아니라 **실질적 의사결정 지원**이 됩니다.
* **풀스택 어필 포인트:** 백엔드 데이터 처리 → AI 요약 → 외부 문서 API 연동까지 풀스택 역량을 보여줍니다.
* **면접관이 열광하는 이유:** “이 개발자는 관리자의 **시간 절약 포인트**를 정확히 짚었네”라는 반응을 얻을 수 있습니다.


차별성 ①: 단순 텍스트 요약을 넘어선 '정량적 지표 분석' (엔지니어링의 차별화)
상용 툴: 주로 작성된 텍스트(태스크 내용)를 기반으로 자연어 요약만 해줍니다.

나의 프로젝트: 단순 텍스트 요약뿐만 아니라, 데이터베이스의 정량적 데이터(태스크 완료율, 평균 소요 시간, 일정 초과 빈도)를 통계적으로 계산(SQL/백엔드 로직)하여 AI 프롬프트에 주입(Context Injection)합니다.

예시: "이번 주에 A 개발자가 담당한 태스크의 평균 소요 시간이 지난주 대비 20% 증가했습니다. 백엔드 파트의 병목이 의심됩니다"와 같이 [통계 데이터 + AI 자연어 분석]이 결합된 보고서를 생성합니다. (단순 AI API 호출이 아니라 백엔드 개발자로서의 데이터 가공 능력을 어필하는 포인트입니다.)

차별성 ②: 다중 플랫폼 연동 및 포맷 커스텀 (Google Docs API의 적극적 활용)
상용 툴: 노션이나 Jira 내부 대시보드에서만 요약본을 볼 수 있는 경우가 많습니다. 외부로 반출하려면 복사/붙여넣기를 해야 합니다.

나의 프로젝트: 실제 회사에서는 보고용 문서 포맷(양식)이 정해져 있습니다. Google Docs API를 활용하여 회사의 공식 보고서 템플릿(표, 타이틀, 레이아웃)에 맞춰 AI가 생성한 내용을 정확한 위치에 바인딩(Binding)하여 문서 파일(.docx 또는 PDF)로 즉시 다운로드 받거나 메일로 발송할 수 있게 만듭니다. '진짜 실무 프로세스'를 고려한 기능입니다.

차별성 ③: 역할별(Role-based) 맞춤형 요약 보고서 분기
상용 툴: 모두에게 똑같은 요약 내용을 보여줍니다.

나의 프로젝트: 대상을 선택할 수 있습니다. 팀장용 보고서(전체 일정 리스크와 거시적 진행률 위주)와 개발자용 보고서(내가 해결해야 할 기술적 이슈와 다음 주 작업 위주)로 프롬프트를 분기하여, 하나의 태스크 데이터를 가지고도 보는 사람의 역할에 맞는 맞춤형 보고서를 생성해 줍니다.
