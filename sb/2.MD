	<select id="weeklyReport" resultType="WeeklyReportDto" parameterType="int">
	<![CDATA[
	select
	p.pro_id as proId, //프로젝트 번호
	p.pro_name as projectName, //프로젝트 이름
	p.end_date as endDate, //끝난 날짜 
	nvl(count(t.task_id), 0) as totalTask, //태스크 총합 nvl -> null이면 0 으로 치환
	nvl(sum(case when t.task_status = 'DONE' // case when 조건 then 1 end : 조건을 만족하면 row는1,아니면 null 근데 nvl로 인해 0 으로 치환 이유? 
                                             //: 매칭이 안되면 sum은 null이 되므로 0으로 바꿔줌 -> 값을 '0~n'에서만 보여주기 위함, null이라는 특수값 거르기 위함
	    and t.actual_end_date >= trunc(sysdate,'IW') - 7 //오늘 날짜를 이번주 월요일 00:00으로 잘라줌 -7은 지난주 월요일
	    and t.actual_end_date <  trunc(sysdate,'IW')     // 상태가 done이고, 실제완료일이 이번주 시작일보다 7일전 이후이고, 이번주 시작일 이전일경우
	    then 1 end), 0) as completedThisWeek,            // 결론: 조건은 "완료일이 지난주 월요일~이번주 월요일 전 날 사이 지난주 한주간 완료된 task를 찾기 위한 조건"
    nvl(sum(case when t.task_status <> 'DONE' and t.task_end_date < sysdate then 1 end), 0) as delayTaskCount, //상태가 done이 아니고,task_end_date가 오늘보다 과거 -> 기한이 지났는데 아직 안끝난 task개수 : 지연 태스크
	nvl(round(sum(case when t.task_status = 'DONE' then 1 else 0 end) / count(t.task_id) * 100), 0) as progressRate, //전체 태스크 완료비율
	nvl(round(avg(case when t.actual_start_date is not null and t.actual_end_date is not null
	    then t.actual_end_date - t.actual_start_date end), 1), 0) as avgTaskDays, //태스크들의 평균 소요일수 계산 소수점 첫째자리까지 반올림
	nvl(round(avg(case when t.actual_end_date > t.task_end_date
	    then t.actual_end_date - t.task_end_date end)), 0) as avgDelayDays //지연된 태스크들의 평균 지연 일수
	from project p
	left join task t on p.pro_id = t.pro_id //해당 프로젝트에 태스크가 없어도 project row는 결과에 남음
	where p.pro_id = #{proId}
	group by p.pro_id, p.pro_name, p.end_date
	]]>
	</select>
